Q4 - IPC Pipes (Likely Variation)
Predicted: Bidirectional or parent reads from child (reverse of Sec-B)

Child writes a message to pipe
Parent reads and displays with its PID

#include <iostream>
#include <unistd.h>
#include <sys/types.h>
#include <cstring>
using namespace std;

int main() {
    int fd[2];
    pid_t pid;
    
    if (pipe(fd) == -1) {
        perror("pipe failed");
        return 1;
    }
    
    pid = fork();
    
    if (pid < 0) {
        perror("fork failed");
        return 1;
    }
    
    if (pid == 0) {  // Child process
        close(fd[0]);
        const char *msg = "Hello from child!";
        write(fd[1], msg, strlen(msg) + 1);
        cout << "Child (PID " << getpid() << "): Message sent" << endl;
        close(fd[1]);
    }
    else {  // Parent process
        close(fd[1]);
        char buffer[100];
        read(fd[0], buffer, sizeof(buffer));
        cout << "Parent (PID " << getpid() << "): Received = " << buffer << endl;
        close(fd[0]);
    }
    
    return 0;
}
