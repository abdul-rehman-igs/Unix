Write a C++ program to perform the following tasks:
. Open an existing file named data.txt in read-only mode.
· Read the first 20 bytes from the file into a buffer.
· Print the contents of the buffer to the screen.
· Use the return value of the read function to ensure that only the number of bytes actually
  read are printed.

nano data.txt

Hello, this is my first file. My name is Abdul REhman Khalid.

nano readfile.cpp

#include <iostream>
#include <fcntl.h>
#include <unistd.h>
#include <cstring>

using namespace std;

int main()
{
    int fd;
    char buffer[21];
    ssize_t bytesRead;

    fd = open("data.txt", O_RDONLY);

    if (fd == -1)
    {
        perror("Error opening file");
        return 1;
    }

    memset(buffer, 0, sizeof(buffer));

    bytesRead = read(fd, buffer, 20);

    if (bytesRead == -1)
    {
        perror("Error reading file");
        close(fd);
        return 1;
    }

    cout << "Bytes read: " << bytesRead << endl;
    cout << "Data: ";

    for (int i = 0; i < bytesRead; i++)
    {
        cout << buffer[i];
    }

    cout << endl;

    close(fd);

    return 0;
}

g++ readfile.cpp -o readfile
./readfile
